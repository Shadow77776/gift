    function checkPassword() {
        const input = document.getElementById('password').value;
        if (input === config.password) {
            document.getElementById('login-page').classList.add('hidden');
            document.getElementById('content-page').classList.remove('hidden');
            document.getElementById('music-btn').classList.remove('hidden');
            
            // Start Music
            audio.play().catch(e => console.log("Autoplay blocked, user must click button first."));
        } else {
            alert("Wrong password!");
        }
    }

    function toggleMusic() {
        if (audio.paused) {
            audio.play();
        } else {
            audio.pause();
        }
    }

    function showMemories() {
        const list = document.getElementById('memories-list');
        list.innerHTML = ""; 
        config.memories.forEach(mem => {
            const div = document.createElement('div');
            div.className = 'memory-item';
            div.innerHTML = `
                <img src="${mem.image}" alt="Memory">
                <p class="message-text">${mem.text}</p>
            `;
            list.appendChild(div);
        });
        list.classList.remove('hidden');
        list.scrollIntoView({ behavior: 'smooth' });
    }
